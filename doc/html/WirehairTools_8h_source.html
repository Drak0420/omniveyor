91 class PCGRandom
92 {
93 public:
94  void Seed(uint64_t y, uint64_t x = 0)
95  {
96  State = 0;
97  Inc = (y << 1u) | 1u;
98  Next();
99  State += x;
100  Next();
101  }
102 
103  GF256_FORCE_INLINE uint32_t Next()
104  {
105  const uint64_t oldstate = State;
106  State = oldstate * UINT64_C(6364136223846793005) + Inc;
107  const uint32_t xorshifted = (uint32_t)(((oldstate >> 18) ^ oldstate) >> 27);
108  const uint32_t rot = oldstate >> 59;
109  return (xorshifted >> rot) | (xorshifted << ((uint32_t)(-(int32_t)rot) & 31));
110  }
111 
112  uint64_t State = 0, Inc = 0;
113 };

225 GF256_FORCE_INLINE void IterateNextColumn(
226  uint16_t &x,
227  const uint16_t b,
228  const uint16_t p,
229  const uint16_t a
230 )
231 {
232  x = (x + a) % p;
233 
234  // If the result is in the range [b, p):
235  if (x >= b)
236  {
237  const uint16_t distanceToP = p - x;
238 
239  // If adding again would wrap around:
240  if (a >= distanceToP) {
241  x = a - distanceToP;
242  }
243  else {
244  // Handle rare case where multiple adds are needed
245  x = (((uint32_t)a << 16) - distanceToP) % a;
246  }
247  }
248 }
