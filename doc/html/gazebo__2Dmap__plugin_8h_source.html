187 template <typename T>
188 class FirstOrderFilter {
189 
190  public:
191  FirstOrderFilter(double timeConstantUp, double timeConstantDown, T initialState):
192  timeConstantUp_(timeConstantUp),
193  timeConstantDown_(timeConstantDown),
194  previousState_(initialState) {}
195 
197  T updateFilter(T inputState, double samplingTime) {
198 
199  T outputState;
200  if (inputState > previousState_) {
201  // Calcuate the outputState if accelerating.
202  double alphaUp = exp(-samplingTime / timeConstantUp_);
203  // x(k+1) = Ad*x(k) + Bd*u(k)
204  outputState = alphaUp * previousState_ + (1 - alphaUp) * inputState;
205 
206  }
207  else {
208  // Calculate the outputState if decelerating.
209  double alphaDown = exp(-samplingTime / timeConstantDown_);
210  outputState = alphaDown * previousState_ + (1 - alphaDown) * inputState;
211  }
212  previousState_ = outputState;
213  return outputState;
214 
215  }
216 
217  ~FirstOrderFilter() {}
218 
219  protected:
220  double timeConstantUp_;
221  double timeConstantDown_;
222  T previousState_;
223 };

226 template<class Derived>
227 Eigen::Quaternion<typename Derived::Scalar> QuaternionFromSmallAngle(const Eigen::MatrixBase<Derived> & theta) {
228  typedef typename Derived::Scalar Scalar;
229  EIGEN_STATIC_ASSERT_FIXED_SIZE(Derived);
230  EIGEN_STATIC_ASSERT_VECTOR_SPECIFIC_SIZE(Derived, 3);
231  const Scalar q_squared = theta.squaredNorm() / 4.0;
232 
233  if (q_squared < 1) {
234  return Eigen::Quaternion<Scalar>(sqrt(1 - q_squared), theta[0] * 0.5, theta[1] * 0.5, theta[2] * 0.5);
235  }
236  else {
237  const Scalar w = 1.0 / sqrt(1 + q_squared);
238  const Scalar f = w * 0.5;
239  return Eigen::Quaternion<Scalar>(w, theta[0] * f, theta[1] * f, theta[2] * f);
240  }
241 }
